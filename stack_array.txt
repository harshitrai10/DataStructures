void Push(int stck[],int *top,int item){
    if(*top == 9){
        printf("OVERFLOW\n");
        // return;
    }
    *top = *top +1;
    stck[*top] = item;
    return;
}
void traverse(int stck[],int top){
    if(top == -1)
    {
        printf("Underflow");
        return;
    }
    int to = top;
    while(to != -1){
        printf("%d ",stck[to]);
        to = to-1;
    }
    return;
}
int pop(int stck[],int *top){
    int item;
    if(*top == -1)
    {
        printf("Underflow");
        return -1;
    }
    item = stck[*top];
    *top = *top -1;
    return item;
}
